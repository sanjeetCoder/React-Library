## Install

```bash
npm install --save @reactlib/test
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from '@reactlib/test'
import '@reactlib/test/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
